function create_portfolio_item() {
    register_post_type( 'portfolio_items',
        array(
            'labels' => array(
                'name' => 'Portfolio Items',
                'singular_name' => 'Portfolio Item',
                'add_new' => 'Add New',
                'add_new_item' => 'Add New Portfolio Item',
                'edit' => 'Edit',
                'edit_item' => 'Edit Portfolio Item',
                'new_item' => 'New Portfolio Item',
                'view' => 'View',
                'view_item' => 'View Portfolio',
                'search_items' => 'Search Portfolio',
                'not_found' => 'No Portfolio Items found',
                'not_found_in_trash' => 'No Portfolio Items found in Trash',
                'parent' => 'Portfolio'
            ),

            'public' => true,
            'menu_position' => 15,
            'supports' => array( 'title', 'editor', 'comments', 'thumbnail'/*, 'custom-fields'*/ ),
            'taxonomies' => array( '' ),
            'menu_icon' => 'dashicons-images-alt2',
            'has_archive' => true
        )
    );
}

function my_admin() {
    add_meta_box( 'portfolio_item_meta_box',
        'Portfolio Item Details',
        'display_portfolio_item_meta_box',
        'portfolio_item', 'normal', 'high'
    );
}
?>

<?php
function display_portfolio_item_meta_box( $portfolio_item ) {
    // Retrieve current name of the Director and Movie Rating based on review ID
    $portfolio_item_client_name = esc_html( get_post_meta( $portfolio_item->ID, 'client_name', true ) );
    $portfolio_item_company_name = intval( get_post_meta( $portfolio_item->ID, 'company_name', true ) );
    ?>
    <div class="postbox">
        <h3>Who</h3>
        <h2>Client</h2>
        <div>
            <input type="text" name="portfolio_item_client_name" value="<?php echo $client_name; ?>" />
        </div>
        <h2>Company</h2>
        <div>
            <input type="text" name="portfolio_item_company_name" value="<?php echo $company_name; ?>" />
        </div>
    </div>
    <?php
}
?>

<?php
function add_portfolio_item_fields( $portfolio_item_id, $portfolio_item ) {
    // Check post type for movie reviews
    if ( $portfolio_item->post_type == 'portfolio_items' ) {
        // Store data in post meta table if present in post data
        if ( isset( $_POST['portfolio_item_client_name'] ) && $_POST['portfolio_item_client_name'] != '' ) {
            update_post_meta( $portfolio_item_id, 'client_name', $_POST['portfolio_item_client_name'] );
        }
        if ( isset( $_POST['portfolio_item_company_name'] ) && $_POST['portfolio_item_company_name'] != '' ) {
            update_post_meta( $portfolio_item_id, 'company_name', $_POST['portfolio_item_company_name'] );
        }
    }
}
?>

<?php
function include_template_function( $template_path ) {
    if ( get_post_type() == 'portfolio_items' ) {
        if ( is_single() ) {
            // checks if the file exists in the theme first,
            // otherwise serve the file from the plugin
            if ( $theme_file = locate_template( array ( 'single-portfolio_items.php' ) ) ) {
                $template_path = $theme_file;
            } else {
                $template_path = plugin_dir_path( __FILE__ ) . '/single-portfolio_items.php';
            }
        }
    }
    return $template_path;
}
?>

<?php
add_action( 'save_post', 'add_portfolio_item_fields', 10, 2 );
add_action( 'admin_init', 'my_admin' );
add_action( 'init', 'create_portfolio_item' );
add_filter( 'template_include', 'include_template_function', 1 );
?>
